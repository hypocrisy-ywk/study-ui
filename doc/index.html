# 安装 - Installation

# 下载

yarn add @hypocrisy-ywk/study-ui

# or

npm i  @hypocrisy-ywk/study-ui

# 使用

在入口文件引入css样式:

  import "@hypocrisy-ywk/study-ui/dist/study-ui.css"

在组件里引用study-ui:

  import {Button} from "@hypocrisy-ywk/study-ui"
